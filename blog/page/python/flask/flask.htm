## 설치 $ pip install flask ## 버전 확인 $ flask --version ## 서버 실행 ( default port : 5000) $ flask run ## 기본예제 : app.py from flask import Flask, render_template app = Flask(__name__) # app = Flask(__name__ , static_url_path='/static' ) @app.route('/') def index(): # return 'Hello Flask' return render_template('index.html') # templates 디렉토리 파일 가져옴 @app.route('/info') def info(): # return 'Info' return render_template('info.html') host_addr = '0.0.0.0' port_num = '8080' if __name__ == "__main__": app.run(host=host_addr, port=port_num) ## werkzeug : 암호화 모듈 (비밀번호) from werkzeug.security import generate_password_hash, check_password_hash hash = generate_password_hash('password') check_password_hash(hash, 'password') # True check_password_hash(hash, 'wrong password') # False
