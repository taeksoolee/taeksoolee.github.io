## arrow function ## generator 함수 * 이터러블을 생성하는 함수 * '이터레이션 프로토콜'을 준수해 이터러블을 생성하는 방식보다 간편하게 이터러블을 구현 # 무한 이터러블 구현 // 이터레이션 프로토콜 const createInfinityByIteration = function() { let i = 0; return { [Symbol.iterator]() { return this; } next () { return { value: ++i }; } } } for(const n of createInfinityByIteration()) { if( n > 5) break; console.log(n); } // 제너레레이터 함수 function* createInfinityByGenerator() { let i = 0; while ( true ) { yield ++i } } for(const n of createInfinityByIteration()) { if( n > 5) break; console.log(n); }
