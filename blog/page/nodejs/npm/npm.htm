## npm ( node package manager ) # 사용 $ npm install [패키지 명] $ npm install --save [패키지명] $ npm install --save-dev [패키지명] // 전역 설치 ex) express-generator, @vue/cli, @angular/cli 등 $ npm install --global [패키지명] $ npm uninstall --global [패키지명] $ npm list [검색어] ## 전역설치 패키지 (업데이트 중...) $ npm install --global typescript $ npm install --global express-generator $ npm install --global @vue/cli $ npm install --global create-react-app $ npm install --global @angular/cli $ npm install --global @nest/cli $ npm install --global node-sass
